```

#!/usr/bin/env python3

# lists [] - These are Data Structures, They are Mutable

movies = ["Hangover", "The Perkins", "Spiderman 3"]

print(movies[1]) # returns the second item
print(movies[0]) # returns the first item

print(movies[1:4])
print(movies[1:])
print(movies[:2])
print(movies[-1])

print(len(movies))

movies.append("JAWS")
print(movies)

movies.pop()
print(movies)

movies.pop(0)
print(movies)

```
